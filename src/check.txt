        <div id="result"></div>

    <script>
        async function login() {
            const username = document.getElementById('username').value;
            const password = document.getElementById('password').value;
            const resultDiv = document.getElementById('result');
            
            resultDiv.innerHTML = 'מתחבר...';
            
            try {
                const response = await fetch('https://giraffeclaude-1.onrender.com/auth/login', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ username, password })
                });
                
                const data = await response.json();
                
                if (response.ok) {
                    resultDiv.innerHTML = 'התחברות הצליחה! ' + JSON.stringify(data);
                } else {
                    resultDiv.innerHTML = 'שגיאה: ' + (data.detail || 'שגיאה לא ידועה');
                }
            } catch (error) {
                resultDiv.innerHTML = 'שגיאת חיבור: ' + error.message;
            }
        }
    </script>
